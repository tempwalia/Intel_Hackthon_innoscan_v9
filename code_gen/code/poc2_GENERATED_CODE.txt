class AudioFile(models.Model):
    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    filename = models.CharField(max_length=250, unique=True)
    content_type = models.CharField(max_length=100)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    def __str__(self):
        return self.filename

class Summary(models.Model):
    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    text = models.TextField()
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    def __str__(self):
        return f"{self.text[:50]}..."

class Action(models.Model):
    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    summary = models.ForeignKey(Summary, on_delete=models.CASCADE)
    timestamp = models.DateTimeField(auto_now_add=True)
    description = models.TextField()

    def __str__(self):
        return f"{self.summary} - {self.timestamp}"

class Minute(models.Model):
    id = models.UUIDField(primary_key=True, default=uuid.uuid4, editable=False)
    summary = models.ForeignKey(Summary, on_delete=models.CASCADE)
    date = models.DateField()
    participants = models.TextField()
    xml_content = models.TextField()
    csv_content = models.TextField()

    def __str__(self):
        return f"Generated Minute for {self.date}"

import React, { useState } from 'react';
import axios from 'axios';
import './App.css';
import UploadButton from './components/UploadButton';

function App() {
    const [audioFileId, setAudioFileId] = useState('');
    const [uploading, setUploading] = useState(false);
    
    const handleUpload = async () => {
        setUploading(true);
        try {
            // Simulate uploading to Azure Blob Storage
            await axios.post('https://your-azure-blob-service-url/upload', { fileId: audioFileId });
            console.log('Audio file uploaded successfully');
            setUploading(false);
        } catch (err) {
            console.error(err);
            setUploading(false);
        }
    };

    return (
        <div className="container mt-5">
            <h1>Voice-Based Meeting Minutes Generator</h1>
            <UploadButton onClick={handleUpload} />
            {!uploading && <p>Uploading... Please wait.</p>}
        </div>
    );
}

export default App;

from flask import Blueprint, render_template, redirect, url_for, flash, abort
from werkzeug.utils import secure_filename
from .models import AudioFile, Summary, Action, Minute
from .forms import UploadForm
from ..config import db
from ..utils import generate_uuid

bp = Blueprint('voice_meeting_generator', __name__, template_folder='templates')

@bp.route('/', methods=('GET', 'POST'))
def index():
    if request.method == 'POST':
        form = UploadForm(request.form)
        if form.validate_on_submit():
            audio_file = AudioFile(filename=form.audio_file.data.filename)
            db.session.add(audio_file)
            db.session.commit()
            flash('Audio file uploaded successfully.', 'success')
            return redirect(url_for('.index'))

    else:
        form = UploadForm()

    return render_template('index.html', form=form)

@bp.route('/transcript/<audio_file_id>')
def transcript(audio_file_id):
    audio_file = AudioFile.query.filter_by(id=audio_file_id).first_or_404()
    blob_service_client = BlobServiceClient.from_connection_string(Config.SQLALCHEMY_DATABASE_URI)
    container_client = blob_service_client.get_container_client(Config.AZURE_CONTAINER)
    blob_client = container_client.get_blob_client(blob=f'{audio_file_id}.mp3')
    download_stream = blob_client.download_blob()
    return download_stream.readall(), 200

@bp.route('/minutes/<summary_id>')
def minutes(summary_id):
    summary = Summary.query.filter_by(id=summary_id).first_or_404()
    blob_service_client = BlobServiceClient.from_connection_string(Config.SQLALCHEMY_DATABASE_URI)
    container_client = blob_service_client.get_container_client(Config.AZURE_CONTAINER)
    blob_client = container_client.get_blob_client(blob=f'{summary_id}.xml')
    xml_content = blob_client.download_blob().readall()
    blob_client = container_client.get_blob_client(blob=f'{summary_id}.csv')
    csv_content = blob_client.download_blob().readall()
    return {'xml': xml_content.decode('utf-8'), 'csv': csv_content.decode('utf-8')}, 200
